       // Fungsi untuk menampilkan Fakta Hari Ini
       function displayFactOfTheDay() {
           const todayFact = factsData[Math.floor(Math.random() * factsData.length)];
           factOfTheDayElement.innerHTML = `
               <img src="${todayFact.image}" alt="${todayFact.title}" />
               <h3>${todayFact.title}</h3>
               <p>${todayFact.shortDescription}</p>
           `;
       }

       // Fungsi untuk menampilkan semua fakta di grid
       function displayFactGrid(filteredFacts = factsData) {
           factGridElement.innerHTML = ''; // Bersihkan konten yang ada
           filteredFacts.forEach(fact => {
               const factCard = document.createElement('div');
               factCard.classList.add('fact-card');
               factCard.innerHTML = `
                   <img src="${fact.image}" alt="${fact.title}">
                   <div class="fact-card-content">
                       <h4>${fact.title}</h4>
                       <p>${fact.shortDescription}</p>
                       <span class="category-tag">${fact.category}</span>
                   </div>
               `;
               // Tambahkan event listener untuk menampilkan modal saat kartu diklik
               factCard.addEventListener('click', () => openFactModal(fact));
               factGridElement.appendChild(factCard);
           });
       }

       // Fungsi untuk membuka modal dengan detail fakta
       function openFactModal(fact) {
           modalImage.src = fact.image;
           modalImage.alt = fact.title;
           modalTitle.textContent = fact.title;
           modalDescription.textContent = fact.description;
           modalCategory.textContent = fact.category;
           factModal.style.display = 'flex'; // Gunakan flex untuk centering
           document.body.style.overflow = 'hidden'; // Nonaktifkan scroll body
       }

       // Fungsi untuk menutup modal
       function closeFactModal() {
           factModal.style.display = 'none';
           document.body.style.overflow = ''; // Aktifkan kembali scroll body
       }

       // Event listener untuk tombol tutup modal
       closeModalButton.addEventListener('click', closeFactModal);

       // Tutup modal jika klik di luar konten modal
       factModal.addEventListener('click', (event) => {
           if (event.target === factModal) {
               closeFactModal();
           }
       });

       // Event listener untuk pencarian
       searchInput.addEventListener('keyup', () => {
           const searchTerm = searchInput.value.toLowerCase();
           const filteredFacts = factsData.filter(fact =>
               fact.title.toLowerCase().includes(searchTerm) ||
               fact.description.toLowerCase().includes(searchTerm) ||
               fact.category.toLowerCase().includes(searchTerm)
           );
           displayFactGrid(filteredFacts);
       });

       // Event listener untuk kategori di sidebar dan navigasi
       function handleCategoryClick(event) {
           event.preventDefault(); // Mencegah reload halaman
           const category = event.target.dataset.category;
           if (category) {
               const filteredFacts = factsData.filter(fact =>
                   category === 'Random' ? true : fact.category === category
               );
               displayFactGrid(filteredFacts);
               // Gulir ke bagian kumpulan fakta setelah memfilter
               document.getElementById('factGrid').scrollIntoView({ behavior: 'smooth' });
               // Tutup menu navigasi di mobile setelah klik kategori
               if (navLinks.classList.contains('active')) {
                   navLinks.classList.remove('active');
               }
           }
       }

       document.querySelectorAll('.nav-links a[data-category]').forEach(link => {
           link.addEventListener('click', handleCategoryClick);
       });
       document.querySelectorAll('#sidebarCategories a[data-category]').forEach(link => {
           link.addEventListener('click', handleCategoryClick);
       });

           });
       }
